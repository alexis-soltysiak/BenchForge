- `.stats` property — returns a dict `{"hits": int, "misses": int}`; only `get` and `peek` count toward hits/misses

```python
import threading
from typing import Any, Callable, Dict, Optional


class _Node:
    """Doubly-linked list node."""
    __slots__ = ("key", "value", "prev", "next")

    def __init__(self, key: Any, value: Any) -> None:
        self.key = key
        self.value = value
        self.prev: Optional["_Node"] = None
        self.next: Optional["_Node"] = None


class LRUCache:
    """
    Thread-safe LRU cache backed by a hash map and a doubly-linked list.

    The doubly-linked list is kept in recency order: the node immediately
    after the sentinel head is the LRU entry; the node immediately before
    the sentinel tail is the MRU entry.
    """

    def __init__(self, capacity: int) -> None:
        if capacity < 1:
            raise ValueError(f"capacity must be >= 1, got {capacity}")
        self._capacity = capacity
        self._map: Dict[Any, _Node] = {}
        self._lock = threading.Lock()
        self._hits = 0
        self._misses = 0

        # Sentinel nodes — never stored in _map.
        self._head = _Node(None, None)  # LRU end
        self._tail = _Node(None, None)  # MRU end
        self._head.next = self._tail
        self._tail.prev = self._head

    # ------------------------------------------------------------------
    # Private linked-list helpers (must be called with lock held)
    # ------------------------------------------------------------------

    def _remove(self, node: _Node) -> None:
        """Unlink node from the list."""
        node.prev.next = node.next
        node.next.prev = node.prev
        node.prev = None
        node.next = None

    def _append_mru(self, node: _Node) -> None:
        """Insert node just before the tail (MRU position)."""
        prev = self._tail.prev
        prev.next = node
        node.prev = prev
        node.next = self._tail
        self._tail.prev = node

    def _move_to_mru(self, node: _Node) -> None:
        """Relocate an already-linked node to the MRU end."""
        self._remove(node)
        self._append_mru(node)

    def _evict_lru(self) -> None:
        """Remove the least-recently-used node (just after head)."""
        lru = self._head.next
        if lru is self._tail:
            return  # empty cache — should never happen in normal flow
        self._remove(lru)
        del self._map[lru.key]

    # ------------------------------------------------------------------
    # Public interface
    # ------------------------------------------------------------------

    def get(self, key: Any) -> Any:
        with self._lock:
            node = self._map.get(key)
            if node is None:
                self._misses += 1
                return -1
            self._hits += 1
            self._move_to_mru(node)
            return node.value

    def peek(self, key: Any) -> Any:
        with self._lock:
            node = self._map.get(key)
            if node is None:
                self._misses += 1
                return -1
            self._hits += 1
            # Intentionally do NOT move to MRU.
            return node.value

    def put(self, key: Any, value: Any) -> None:
        with self._lock:
            node = self._map.get(key)
            if node is not None:
                # Update existing entry and mark as MRU.
                node.value = value
                self._move_to_mru(node)
            else:
                # Insert new entry.
                if len(self._map) >= self._capacity:
                    self._evict_lru()
                new_node = _Node(key, value)
                self._map[key] = new_node
                self._append_mru(new_node)

    def put_if_absent(self, key: Any, value: Any) -> bool:
        with self._lock:
            if key in self._map:
                # Key already exists — do nothing, preserve recency.
                return False
            # Insert new entry.
            if len(self._map) >= self._capacity:
                self._evict_lru()
            new_node = _Node(key, value)
            self._map[key] = new_node
            self._append_mru(new_node)
            return True

    def evict_where(self, predicate: Callable[[Any, Any], bool]) -> None:
        with self._lock:
            # Collect nodes to remove first to avoid mutating while iterating.
            to_remove = [
                node
                for node in self._map.values()
                if predicate(node.key, node.value)
            ]
            for node in to_remove:
                self._remove(node)
                del self._map[node.key]

    @property
    def stats(self) -> Dict[str, int]:
        with self._lock:
            return {"hits": self._hits, "misses": self._misses}
```

```python
import threading


class LRUCache:
    class _Node:
        __slots__ = ("key", "value", "prev", "next")

        def __init__(self, key=None, value=None):
            self.key = key
            self.value = value
            self.prev = None
            self.next = None

    def __init__(self, capacity):
        if capacity < 1:
            raise ValueError("capacity must be at least 1")
        self._capacity = capacity
        self._map = {}
        self._head = self._Node()  # sentinel: least recent side
        self._tail = self._Node()  # sentinel: most recent side
        self._head.next = self._tail
        self._tail.prev = self._head
        self._lock = threading.Lock()
        self._hits = 0
        self._misses = 0

    def _add_to_tail(self, node):
        last = self._tail.prev
        last.next = node
        node.prev = last
        node.next = self._tail
        self._tail.prev = node

    def _remove_node(self, node):
        prev_node = node.prev
        next_node = node.next
        prev_node.next = next_node
        next_node.prev = prev_node
        node.prev = None
        node.next = None

    def _move_to_tail(self, node):
        self._remove_node(node)
        self._add_to_tail(node)

    def get(self, key):
        with self._lock:
            node = self._map.get(key)
            if node is None:
                self._misses += 1
                return -1
            self._hits += 1
            self._move_to_tail(node)
            return node.value

    def peek(self, key):
        with self._lock:
            node = self._map.get(key)
            if node is None:
                self._misses += 1
                return -1
            self._hits += 1
            return node.value

    def put(self, key, value):
        with self._lock:
            node = self._map.get(key)
            if node is not None:
                node.value = value
                self._move_to_tail(node)
                return

            if len(self._map) >= self._capacity:
                lru = self._head.next
                self._remove_node(lru)
                del self._map[lru.key]

            node = self._Node(key, value)
            self._map[key] = node
            self._add_to_tail(node)

    def put_if_absent(self, key, value):
        with self._lock:
            if key in self._map:
                return False

            if len(self._map) >= self._capacity:
                lru = self._head.next
                self._remove_node(lru)
                del self._map[lru.key]

            node = self._Node(key, value)
            self._map[key] = node
            self._add_to_tail(node)
            return True

    def evict_where(self, predicate):
        with self._lock:
            node = self._head.next
            while node is not self._tail:
                nxt = node.next
                if predicate(node.key, node.value):
                    self._remove_node(node)
                    del self._map[node.key]
                node = nxt
```
