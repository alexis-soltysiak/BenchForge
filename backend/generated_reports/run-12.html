```python
import threading


class TxnStore:
    _ACTIVE = 0
    _COMMITTED = 1
    _ROLLED_BACK = 2

    def __init__(self):
        self._lock = threading.Lock()
        self._committed = {}              # key -> value
        self._txns = {}                   # txn_id -> {'state': int, 'writes': dict}
        self._snapshots = {}              # snap_id -> dict (immutable copy)
        self._next_txn_id = 0
        self._next_snap_id = 0

    def begin(self):
        with self._lock:
            self._next_txn_id += 1
            tid = self._next_txn_id
            self._txns[tid] = {'state': self._ACTIVE, 'writes': {}}
            return tid

    def write(self, txn_id, key, value):
        with self._lock:
            txn = self._txns.get(txn_id)
            if txn is None or txn['state'] != self._ACTIVE:
                raise KeyError(txn_id)
            txn['writes'][key] = value

    def read(self, txn_id, key):
        with self._lock:
            txn = self._txns.get(txn_id)
            if txn is None or txn['state'] != self._ACTIVE:
                raise KeyError(txn_id)
            if key in txn['writes']:
                return txn['writes'][key]
            if key in self._committed:
                return self._committed[key]
            raise KeyError(key)

    def commit(self, txn_id):
        with self._lock:
            txn = self._txns.get(txn_id)
            if txn is None or txn['state'] != self._ACTIVE:
                raise KeyError(txn_id)
            # Last-writer-wins: simply overwrite committed values.
            for k, v in txn['writes'].items():
                self._committed[k] = v
            txn['state'] = self._COMMITTED
            txn['writes'] = {}

    def rollback(self, txn_id):
        with self._lock:
            txn = self._txns.get(txn_id)
            if txn is None or txn['state'] != self._ACTIVE:
                raise KeyError(txn_id)
            txn['state'] = self._ROLLED_BACK
            txn['writes'] = {}

    def read_committed(self, key):
        with self._lock:
            if key not in self._committed:
                raise KeyError(key)
            return self._committed[key]

    def snapshot(self):
        with self._lock:
            self._next_snap_id += 1
            sid = self._next_snap_id
            # Shallow copy of committed dict; values are treated as immutable references.
            self._snapshots[sid] = dict(self._committed)
            return sid

    def read_snapshot(self, snapshot_id, key):
        with self._lock:
            snap = self._snapshots.get(snapshot_id)
            if snap is None:
                raise KeyError(snapshot_id)
            if key not in snap:
                raise KeyError(key)
            return snap[key]
```

```python
import threading
from typing import Any, Dict


class TxnStore:
    """
    A thread‑safe transactional key‑value store that supports
    begin / write / read / commit / rollback, read‑committed,
    and immutable snapshots.
    """

    # --------------------------------------------------------------------- #
    # Construction & internal helpers
    # --------------------------------------------------------------------- #

    def __init__(self) -> None:
        # Global lock protecting all mutable management structures
        self._global_lock = threading.Lock()

        # Monotonically increasing identifiers
        self._next_txn_id = 1
        self._next_snapshot_id = 1

        # txn_id → {'writes': {k: v}, 'state': 'active'|'committed'|'rolledback'}
        self._txns: Dict[int, Dict[str, Any]] = {}

        # Committed key‑value store
        self._committed: Dict[Any, Any] = {}

        # snapshot_id → copy of committed dict at capture time
        self._snapshots: Dict[int, Dict[Any, Any]] = {}

        # Per‑key locks used only during commit (last‑writer‑wins)
        self._key_locks: Dict[Any, threading.Lock] = {}

    # --------------------------------------------------------------------- #
    # Transaction life‑cycle
    # --------------------------------------------------------------------- #

    def begin(self) -> int:
        """Start a new transaction and return its unique id."""
        with self._global_lock:
            txn_id = self._next_txn_id
            self._next_txn_id += 1
            self._txns[txn_id] = {"writes": {}, "state": "active"}
        return txn_id

    def _check_txn_active(self, txn_id: int) -> Dict[str, Any]:
        """Return the txn dict if it exists and is active, else raise KeyError."""
        with self._global_lock:
            txn = self._txns.get(txn_id)
            if txn is None or txn["state"] != "active":
                raise KeyError(f"invalid or closed transaction {txn_id}")
            return txn

    # --------------------------------------------------------------------- #
    # Write / read within a transaction
    # --------------------------------------------------------------------- #

    def write(self, txn_id: int, key: Any, value: Any) -> None:
        """Write a key‑value pair inside the given active transaction."""
        txn = self._check_txn_active(txn_id)
        # No need for extra locking – the txn dict is only reachable by its owner
        txn["writes"][key] = value

    def read(self, txn_id: int, key: Any) -> Any:
        """
        Read a key as visible to the transaction.
        Own writes shadow the committed value.
        """
        txn = self._check_txn_active(txn_id)

        if key in txn["writes"]:
            return txn["writes"][key]

        with self._global_lock:
            if key in self._committed:
                return self._committed[key]

        raise KeyError(f"key {key!r} not found in transaction {txn_id}")

    # --------------------------------------------------------------------- #
    # Commit / rollback
    # --------------------------------------------------------------------- #

    def commit(self, txn_id: int) -> None:
        """
        Make all writes from the transaction visible to subsequent
        read‑committed operations.  The commit wins over any previous
        commit of the same key (last‑writer‑wins).
        """
        # 1. Validate & capture the writes atomically
        with self._global_lock:
```

```python
    """
    Thread-safe LRU cache backed by a hash map and a doubly-linked list.

    The doubly-linked list maintains recency order in O(1):
      - head.next  → least recently used (LRU)
      - tail.prev  → most recently used  (MRU)

    Sentinel head/tail nodes eliminate all None-checks on boundary operations.

            raise ValueError(f"capacity must be >= 1, got {capacity}")

        self._map: Dict[Any, _Node] = {}

        self._misses = 0

        self._tail = _Node()  # MRU sentinel

        self._tail.prev = self._head

        node.prev.next = node.next
        node.next.prev = node.prev

        node.prev = self._tail.prev
        node.next = self._tail
        self._tail.prev.next = node
        self._tail.prev = node

        self._remove(node)
        self._append_mru(node)
```

```python
from typing import Any, Callable, Dict, Optional


class _Node:
    __slots__ = ("key", "value", "prev", "next")

    def __init__(self, key: Any, value: Any) -> None:
        self.key = key
        self.value = value
        self.prev: Optional["_Node"] = None
        self.next: Optional["_Node"] = None


class LRUCache:
    """Thread‑safe LRU cache with O(1) usual operations."""

    def __init__(self, capacity: int) -> None:
        if capacity < 1:
            raise ValueError("capacity must be >= 1")
        self._capacity = capacity
        self._map: Dict[Any, _Node] = {}
        self._head: Optional[_Node] = None      # LRU
        self._tail: Optional[_Node] = None      # MRU
        self._lock = Lock()
        self._hits = 0
        self._misses = 0

    # --------------------------------------------------------------------- #
    # doubly‑linked list helpers (all called with lock held)
    # --------------------------------------------------------------------- #
    def _add_tail(self, node: _Node) -> None:
        """Insert *node* as most‑recent (rightmost)."""
        if self._tail is None:                # empty list
            self._head = self._tail = node
        else:
            node.prev = self._tail
            self._tail.next = node
            self._tail = node
        node.next = None

    def _remove(self, node: _Node) -> None:
        """Unlink *node* from the list."""
        if node.prev:
            node.prev.next = node.next
        else:                                 # node is head
            self._head = node.next
        if node.next:
            node.next.prev = node.prev
        else:                                 # node is tail
            self._tail = node.prev
        node.prev = node.next = None

    def _move_to_tail(self, node: _Node) -> None:
        """Mark *node* as most‑recent."""
        if node is self._tail:
            return
        self._remove(node)
        self._add_tail(node)

    # --------------------------------------------------------------------- #
    # public API
    # --------------------------------------------------------------------- #
    def get(self, key: Any) -> Any:
        """Return value and promote to MRU, or -1 if absent."""
        with self._lock:
            node = self._map.get(key)
            if node is None:
                self._misses += 1
                return -1
            self._hits += 1
            self._move_to_tail(node)
            return node.value
```

```python
from bisect import insort, bisect_left, bisect_right


class RateLimiter:
    """
    Rolling-window rate limiter: at most 3 requests per user
    in any 10-second window (inclusive).

    Handles out-of-order and duplicate timestamps correctly by
    retaining full per-user history of *accepted* requests.
    """

    WINDOW = 10      # seconds
    LIMIT = 3        # max requests per window

    def __init__(self):
        # user_id -> sorted list of accepted timestamps
        self._history = {}

    def allow(self, user_id, timestamp):
        if not isinstance(timestamp, int):
            raise TypeError("timestamp must be an integer (seconds)")

        history = self._history.setdefault(user_id, [])

        # Define the rolling window as [timestamp - WINDOW + 1, timestamp + WINDOW - 1].
        # For any 10-second window that *contains* `timestamp`, all other
        # timestamps in that window must lie within WINDOW-1 seconds of it.
        # We must ensure no such window already contains LIMIT accepted events.
        low = timestamp - (self.WINDOW - 1)
        high = timestamp + (self.WINDOW - 1)

        # Collect all accepted events within [low, high] plus the candidate.
        left = bisect_left(history, low)
        right = bisect_right(history, high)
        neighbors = history[left:right]

        # Insert candidate into a sorted copy and slide a window of size WINDOW
        # across it. If any contiguous group of LIMIT+1 events fits inside
        # any 10-second window, the candidate must be rejected.
        candidate_list = list(neighbors)
        insort(candidate_list, timestamp)

        # For every group of (LIMIT+1) consecutive events that includes the
        # candidate, check whether they all fit within a WINDOW-second span.
        # If max - min < WINDOW, they fit into some 10-second rolling window.
        n = len(candidate_list)
        for i in range(n - self.LIMIT):
            # window of LIMIT+1 events: indices i .. i+LIMIT
            if candidate_list[i + self.LIMIT] - candidate_list[i] < self.WINDOW:
                # Only reject if this violating group actually contains
                # the candidate (otherwise the violation is historical
                # and was already accepted previously).
                # Since we only inserted the candidate once, and all stored
                # history was previously valid, any new violation must
                # involve the candidate.
                return False

        # Accept: record the timestamp in sorted order.
        insort(history, timestamp)
        return True
```

```python
        """

        Uses binary search for O(log n) boundary finding, O(1) count.

        hi = self._bisect_right(sorted_list, inclusive_end)

                lo = mid + 1
```
