class LRUCache:
    """
    LRU Cache implemented with a doubly-linked list and a hash map.

    The doubly-linked list maintains recency order:
      - head.next  → most recently used
      - tail.prev  → least recently used

    The hash map maps each key to its corresponding list node,
    giving O(1) lookup, insertion, and deletion.
    """

    class _Node:
        __slots__ = ("key", "value", "prev", "next")

        def __init__(self, key=0, value=0):
            self.key = key
            self.value = value
            self.prev = None
            self.next = None

    def __init__(self, capacity: int):
        if capacity < 1:
            raise ValueError("capacity must be a positive integer")
        self._capacity = capacity
        self._map: dict[int, LRUCache._Node] = {}

        # Sentinel nodes — they never hold real data.
        self._head = self._Node()   # dummy head (most-recent side)
        self._tail = self._Node()   # dummy tail (least-recent side)
        self._head.next = self._tail
        self._tail.prev = self._head

    # ------------------------------------------------------------------
    # Internal helpers
    # ------------------------------------------------------------------

    def _remove(self, node: "_Node") -> None:
        """Unlink an existing node from the list in O(1)."""
        node.prev.next = node.next
        node.next.prev = node.prev

    def _insert_front(self, node: "_Node") -> None:
        """Insert a node immediately after the head sentinel in O(1)."""
        node.next = self._head.next
        node.prev = self._head
        self._head.next.prev = node
        self._head.next = node

    def _move_to_front(self, node: "_Node") -> None:
        """Mark a node as most-recently used in O(1)."""
        self._remove(node)
        self._insert_front(node)

    # ------------------------------------------------------------------
    # Public interface
    # ------------------------------------------------------------------

    def get(self, key: int) -> int:
        node = self._map.get(key)
        if node is None:
            return -1
        self._move_to_front(node)

    class _Node:
        __slots__ = ("key", "val", "prev", "next")

        def __init__(self, key: int, val: int):

            self.val = val
            self.prev = None
            self.next = None

    def __init__(self, capacity: int):
        if capacity <= 0:
            raise ValueError("capacity must be a positive integer")
        self._cap = capacity
        self._map: dict[int, LRUCache._Node] = {}

        # dummy head (most‑recent) and tail (least‑recent) nodes
        self._head = self._Node(0, 0)   # most‑recent sentinel
        self._tail = self._Node(0, 0)   # least‑recent sentinel
        self._head.next = self._tail
        self._tail.prev = self._head

    # --------------------------------------------------------------------- #
    # internal helpers for list manipulation
    # --------------------------------------------------------------------- #
    def _remove(self, node: "_Node") -> None:
        """Detach *node* from the linked list."""
        prev, nxt = node.prev, node.next
        prev.next = nxt
        nxt.prev = prev
        node.prev = node.next = None      # help GC / accidental reuse

    def _add_to_front(self, node: "_Node") -> None:
        """Insert *node* right after the head (most‑recent position)."""
        first = self._head.next
        node.prev = self._head
        node.next = first
        self._head.next = node
        first.prev = node

    def _make_recent(self, node: "_Node") -> None:
        """Move an existing node to the most‑recent position."""
        self._remove(node)
        self._add_to_front(node)

    # --------------------------------------------------------------------- #
    # public API
    # --------------------------------------------------------------------- #
    def get(self, key: int) -> int:
        """Return value for *key* and mark it as most‑recent, or -1 if absent."""
        node = self._map.get(key)
        if node is None:
            return -1
        self._make_recent(node)
        return node.val

    def put(self, key: int, value: int) -> None:
        """Insert or update *key* with *value*; evict LRU if needed."""
        if key in self._map:
            node = self._map[key]
            node.val = value          # update value
            self._make_recent(node)   # mark recent
            return

        # new entry
        new_node = self._Node(key, value)
        self._map[key] = new_node
        self._add_to_front(new_node)

        if len(self._map) > self._cap:          # capacity exceeded
            lru = self._tail.prev               # least‑recent node
            self._remove(lru)
            del self._map[lru.key]

        prev_.next = nxt
        nxt.prev = prev_

        first = self._head.next
        node.prev = self._head
        node.next = first
        self._head.next = node
        first.prev = node

        self._add_to_front(node)

    Rate limiter allowing at most 3 requests per user in any rolling
    10-second window. Handles out-of-order timestamps correctly.
    """

    WINDOW = 10  # seconds

        # user_id -> sorted list of accepted request timestamps
        self._history = {}

    def allow(self, user_id, timestamp):
        if not isinstance(timestamp, int):
            raise TypeError("timestamp must be an integer number of seconds")

        timestamps = self._history.setdefault(user_id, [])

        # A rolling 10-second window containing `timestamp` is any window
        # [w, w + 10) such that w <= timestamp < w + 10, i.e.
        # w in (timestamp - 10, timestamp]. For the new request to be
        # allowed, EVERY such window must contain at most MAX_REQUESTS
        # requests (including this one).
        #
        # Equivalently: for every accepted timestamp t with
        # |t - timestamp| < 10, count how many accepted timestamps fall
        # in the window [min(t, timestamp), min(t, timestamp) + 10).
        # The new request is allowed iff no such window would exceed
        # MAX_REQUESTS after insertion.
        #
        # A simpler sufficient-and-necessary check: slide a window of
        # length WINDOW across every candidate left-endpoint. Candidate
        # endpoints only need to be actual timestamps (existing ones
        # and the new one), because window counts only change at those
        # points.

        # Build the hypothetical set of timestamps if we accept this one.
        # We won't actually insert yet.
        candidates = sorted(set(timestamps + [timestamp]))

        for left in candidates:
            # Only consider windows that contain the new timestamp,
            # since windows not containing it are unaffected by this
            # request and were valid before.
            if left > timestamp:
                break
            if timestamp - left >= self.WINDOW:
                continue  # window [left, left+WINDOW) does not contain timestamp

            right = left + self.WINDOW  # exclusive
            # Count accepted timestamps in [left, right) plus the new one.
            lo = bisect_left(timestamps, left)
            hi = bisect_left(timestamps, right)
            count_existing = hi - lo
            if count_existing + 1 > self.MAX_REQUESTS:
                return False

        # Accept: insert in sorted order.
        insort(timestamps, timestamp)
        return True

    - At most 3 requests are allowed in any 10‑second interval.
    - Timestamps are integer seconds and may arrive out of order.
    """

        # For each user keep a *sorted* list of all request timestamps seen so far.
        self._users: Dict[int, List[int]] = defaultdict(list)

    def allow(self, user_id: int, timestamp: int) -> bool:
        """
        Record a request made by `user_id` at `timestamp` and decide whether it is
        allowed.

        Parameters
        ----------
        user_id : int
            Identifier of the user making the request.
        timestamp : int
            Timestamp of the request in seconds (may be out of order).

        Returns
        -------
        bool
            True if the request does not exceed the limit of 3 requests in the
            inclusive window [timestamp‑9, timestamp]; otherwise False.
        """
        # 1. Insert the new timestamp while keeping the list sorted.
        timestamps = self._users[user_id]
        insort(timestamps, timestamp)          # O(log n) insertion

        # 2. Find how many timestamps fall inside the 10‑second rolling window.
        left = bisect_left(timestamps, timestamp - 9)   # inclusive start
        right = bisect_right(timestamps, timestamp)    # inclusive end

        return cnt <= 3
