from bisect import bisect_left, bisect_right, insort
from collections import defaultdict

class RateLimiter:
    """
    Allows at most 3 requests per user in any rolling 10‑second window.
    Timestamps are integer seconds and may arrive out of order.
    """
    def __init__(self):
        # For each user keep a *sorted* list of all timestamps seen so far.
        self._records = defaultdict(list)          # user_id -> sorted list[int]

    def allow(self, user_id, timestamp: int) -> bool:
        """
        Return True if the request should be allowed, otherwise False.

        The request timestamp is inserted into the user's history
        before the limit check so that later queries see the full set
        of attempts (allowed or denied).
        """
        if not isinstance(timestamp, int):
            raise TypeError("timestamp must be an integer number of seconds")

        # ------------------------------------------------------------
        # 1. Insert the new timestamp while keeping the list sorted.
        # ------------------------------------------------------------
        user_ts = self._records[user_id]
        insort(user_ts, timestamp)          # O(log n)

        # ------------------------------------------------------------
        # 2. Count how many timestamps fall inside the 10‑second window
        #    that ends at the current request time.
        #    Window is inclusive: [timestamp‑9, timestamp]
        # ------------------------------------------------------------
        window_start = timestamp - 9        # 10 seconds inclusive
        left_idx  = bisect_left(user_ts, window_start)   # first >= start
        right_idx = bisect_right(user_ts, timestamp)     # first > end
        cnt = right_idx - left_idx

        # ------------------------------------------------------------
        # 3. Decide.
        # ------------------------------------------------------------
        return cnt <= 3
```

        at most 3 requests in any rolling 10-second window.

        Window interpretation:
        [timestamp - 9, timestamp] inclusive
        so timestamps exactly 10 seconds apart are NOT in the same window.
        """
        if user_id not in self._allowed:
            self._allowed[user_id] = [timestamp]
            return True

        times = self._allowed[user_id]
        left = timestamp - 9
        right = timestamp + 9

        # Count allowed requests already stored that fall within
        # the only range relevant to deciding whether adding `timestamp`
        # would create a violating 10-second window.
        #
        # A prior allowed time `t` matters iff there exists some 10-second
        # window containing both `timestamp` and `t`, which is equivalent to
        # |t - timestamp| <= 9.
        nearby_count = 0
        for t in times:
            if left <= t <= right:
                nearby_count += 1

        # timestamp, then adding this one would create some 10-second window
        # containing 4 requests.

        i = self._bisect_right(times, timestamp)
        times.insert(i, timestamp)
        return True

            mid = (lo + hi) // 2
            if x < arr[mid]:
                hi = mid
            else:
                lo = mid + 1
        return lo

class LRUCache:
    """Least‑Recently‑Used cache with O(1) get / put."""

    class _Node:
        __slots__ = ("key", "val", "prev", "next")
        def __init__(self, key=None, val=None):
            self.key = key
            self.val = val
            self.prev = self.next = None

    def __init__(self, capacity: int):
        if capacity <= 0:
            raise ValueError("capacity must be a positive integer")
        self._capacity = capacity
        self._lookup = {}                     # key → _Node
        # dummy head (most‑recent) and tail (least‑recent)
        self._head = self._Node()
        self._tail = self._Node()
        self._head.next = self._tail
        self._tail.prev = self._head

    # -----------------------------------------------------------------
    # internal helpers (all O(1))
    def _remove(self, node: "_Node") -> None:
        """Detach node from the linked list."""
        node.prev.next = node.next
        node.next.prev = node.prev

    def _add_to_front(self, node: "_Node") -> None:
        """Insert node right after head → marks it most‑recent."""
        node.next = self._head.next
        node.prev = self._head
        self._head.next.prev = node
        self._head.next = node

    def _move_to_front(self, node: "_Node") -> None:
        """Refresh node's recency."""
        self._remove(node)
        self._add_to_front(node)

    # -----------------------------------------------------------------
    def get(self, key: int) -> int:
        """Return value if key exists, otherwise -1.  Marks key MRU."""
        node = self._lookup.get(key)
        if node is None:
            return -1
        self._move_to_front(node)
        return node.val

    def put(self, key: int, value: int) -> None:
        """Insert a new key‑value pair or update an existing one.
        If capacity is exceeded, evicts the LRU key."""
        node = self._lookup.get(key)

        if node:                     # key already present → update
            node.val = value
            self._move_to_front(node)
            return

        # key is new
        if len(self._lookup) >= self._capacity:
            # evict LRU: node just before dummy tail
            lru = self._tail.prev
            self._remove(lru)
            del self._lookup[lru.key]

        new_node = self._Node(key, value)
        self._lookup[key] = new_node
        self._add_to_front(new_node)
```

**Why each operation is O(1):**  
All stored data are either in a hash table (`_lookup`) for constant‑time key lookup, or in a doubly‑linked list where insertion, removal and moving a node involve only a fixed number of pointer updates. `get` performs a hash lookup plus at most one node move; `put` does a hash lookup, possibly one eviction (constant‑time removal), and one insertion, each using only constant‑time steps. No loops over the cache size are needed, guaranteeing average‑case O(1) complexity.

class LRUCache:
    def __init__(self, capacity: int):
        self.capacity = capacity
        self.cache = {}
        self.size = 0
        
        self.head = Node()  # dummy head (most recently used)
        self.tail = Node()  # dummy tail (least recently used)
        self.head.next = self.tail
        self.tail.prev = self.head

    def _add_to_head(self, node: Node) -> None:
        node.prev = self.head
        node.next = self.head.next
        self.head.next.prev = node
        self.head.next = node

    def _remove_node(self, node: Node) -> None:
        node.prev.next = node.next
        node.next.prev = node.prev

    def _move_to_head(self, node: Node) -> None:
        self._remove_node(node)
        self._add_to_head(node)

    def _pop_tail(self) -> Node:
        node = self.tail.prev
        self._remove_node(node)
        return node

    def get(self, key: int) -> int:
        if key not in self.cache:
            return -1
        node = self.cache[key]
        self._move_to_head(node)
        return node.value

    def put(self, key: int, value: int) -> None:
        if key in self.cache:
            node = self.cache[key]
            node.value = value
            self._move_to_head(node)
        else:
            node = Node(key, value)
            self.cache[key] = node
            self._add_to_head(node)
            self.size += 1
            if self.size > self.capacity:
                tail_node = self._pop_tail()
                del self.cache[tail_node.key]
                self.size -= 1
